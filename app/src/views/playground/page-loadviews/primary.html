This example tests Page::loadViews(viewDefs, isAsync).
Notice that if isAsync=false it return the error Element [data-rg-view="#part1"] not found.. To fix the error all views must be compiled in the regoch.json. 

/*** CONTROLLER Page_LoadViewsCtrl.js ***/
await this.loadViews([
  ['#primary', 'playground/page-loadviews/primary.html', 'sibling'], // loaded in the index.html
  ['#primary.html#part1', 'playground/page-loadviews/part1.html'], // loaded in the primary.html
  ['#primary.html#part2', 'playground/page-loadviews/part2.html', 'append'], // loaded in the primary.html
  ['#primary.html#part3', 'playground/page-loadviews/part3.html', 'prepend'] // loaded in the primary.html
], true);
  

/*** VIEW primary.html ***/
<h1>Page::loadViews() Test</h1>
